get_alpha = function(scale){
alpha = data |> 
  select(codebook |> filter(Unique_Alpha == scale) |> pull(Item)) |>  
  psych::alpha(keys = codebook |> filter(Unique_Alpha == scale) |> pull(Inv))
return(list(
  "alpha" = alpha$total$std.alpha,
  "cors" = alpha$item.stats$r.drop,
  "scores" = alpha$scores
))
}

# Perform Kolmogorov-Smirnov test
alphas_ks_test <- alphas %>%
  select(value, scores) %>%
  unnest(scores) %>%
  group_by(value) %>%
  nest() %>%
  mutate(sw_test = map(data, ~shapiro.test(.x$scores))) %>%
  mutate(p_value = map_dbl(sw_test, "p.value"),
         p_value = round(p_value, 3)) |> 
  mutate(value = str_replace(value, "_","\n"))

# Create the plot
alphas_ks_test %>%
  unnest(data) %>%
  ggplot(aes(scores)) +
