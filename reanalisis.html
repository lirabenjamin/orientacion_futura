get_alpha = function(scale){
alpha = data |> 
  select(codebook |> filter(Unique_Alpha == scale) |> pull(Item)) |>  
  psych::alpha(keys = codebook |> filter(Unique_Alpha == scale) |> pull(Inv))
return(list(
  "alpha" = alpha$total$std.alpha,
  "cors" = alpha$item.stats$r.drop,
  "scores" = alpha$scores
))
}

alphas = alphas |>  
  enframe() |> 
  mutate(alpha = map(value, get_alpha),
         alfa = map_dbl(alpha, "alpha"),
         cors = map(alpha, "cors"),
         scores = map(alpha, "scores"))

# Perform Kolmogorov-Smirnov test
alphas_ks_test <- alphas %>%
  select(value, scores) %>%
  unnest(scores) %>%
  group_by(value) %>%
  nest() %>%
  mutate(sw_test = map(data, ~shapiro.test(.x$scores))) %>%
  mutate(p_value = map_dbl(sw_test, "p.value"),
         p_value = round(p_value, 3)) |> 
  mutate(value = str_replace(value, "_","\n"))

# Create the plot
alphas_ks_test %>%
  unnest(data) %>%
  ggplot(aes(scores)) +
  geom_density(aes(y = ..scaled.. * max(..density..)), fill = "gray40", alpha = .3, color = NA) +
  geom_histogram(aes(y = ..density..), bins = 30, alpha = .6) +
  facet_wrap(~value, nrow = 3, scales = "free_y") +
  geom_text(aes(label = ifelse(p_value > 0, paste0("p = ", p_value), "p < .001"), x = 5, y = Inf), hjust = .5, vjust = 2.5, check_overlap = TRUE, data = alphas_ks_test, size = 3)+
